    K-Modes¶

    Source data : https://data.world/uci/balance-scale/workspace/file?filename=balance-scale.data.csv

    Tipe data : Categorical

    Source code 

    import numpy as np
from kmodes.kmodes import KModes

x = np.genfromtxt('balancescale.csv', dtype=int, delimiter=',')[:, :-1]

kmodes_huang = KModes(n_clusters=4, init='Huang', verbose=1)
kmodes_huang.fit(x)

print('k-modes (Balace-scale.csv) centroid:')
print(kmodes_huang.cluster_centroids_)
print('Final training cost: {}'.format(kmodes_huang.cost_))
print('Training Interations: {}'.format(kmodes_huang.n_iter_))

    K-Prototype¶

    Source data : https://data.world/exercises/data-wrangling-exercise-1/workspace/file?filename=Crime_2015.csv

    Tipe data : Categorical & Numerik

    Souce Code

    import numpy as np
from kmodes.kprototypes import KPrototypes

syms = np.genfromtxt('Crime_2015.csv', dtype=str, delimiter=',')[:, 0]
X = np.genfromtxt('Crime_2015.csv', dtype=object, delimiter=',')[:, 1:]
X[:, 0] = X[:, 0].astype(float)

kproto = KPrototypes(n_clusters=2, init='Cao', verbose=2)
clusters = kproto.fit_predict(X, categorical=[9, 2])

print(kproto.cluster_centroids_)
print(kproto.cost_)
print(kproto.n_iter_)

for s, c in zip(syms, clusters):
    print("Symbol: {}, cluster:{}".format(s, c))
